126 def to_doxyfile(self) -> str:
127 """
128 @brief Convert configuration to Doxyfile format
129 @return String containing complete Doxyfile configuration
130
131 @details Generates a complete Doxyfile configuration string that can be
132 written to disk and used with the doxygen command-line tool. The output
133 includes all configured options with appropriate YES/NO values and
134 proper escaping for file paths and patterns.
135
136 @note The generated Doxyfile is compatible with Doxygen 1.9.0+
137 """
138 lines = [
139 f"# Doxyfile generated by Doxygen MCP Server",
140 f"",
141 f"# Project related configuration options",
142 f"PROJECT_NAME = \"{self.project_name}\"",
143 f"PROJECT_NUMBER = \"{self.project_number}\"",
144 f"PROJECT_BRIEF = \"{self.project_brief}\"",
145 f"PROJECT_LOGO = \"{self.project_logo}\"",
146 f"OUTPUT_DIRECTORY = \"{self.output_directory}\"",
147 f"",
148 f"# Build related configuration options",
149 f"EXTRACT_ALL = {'YES' if self.extract_all else 'NO'}",
150 f"EXTRACT_PRIVATE = {'YES' if self.extract_private else 'NO'}",
151 f"EXTRACT_STATIC = {'YES' if self.extract_static else 'NO'}",
152 f"EXTRACT_LOCAL_CLASSES = {'YES' if self.extract_local_classes else 'NO'}",
153 f"",
154 f"# Input related configuration options",

158 ]

366 async def _create_project(self, args: Dict[str, Any]) -> Sequence[TextContent]:
367 """
368 @brief Create a new Doxygen project with intelligent configuration
369 @param args Dictionary containing project parameters
370 @return Sequence of TextContent with creation results
371
372 @details Creates a new Doxygen project by:
373 - Setting up the project directory structure
374 - Generating language-specific configuration
375 - Creating an optimized Doxyfile
376 - Storing project settings for future operations
377
378 @note The generated configuration includes language-specific optimizations
379 and file patterns appropriate for the specified programming language.
380 """
381 project_name = args["project_name"]
382 project_path = Path(args["project_path"])
383 language = args.get("language", "mixed")
384 include_subdirs = args.get("include_subdirs", True)
385 extract_private = args.get("extract_private", False)
386
387 try:
388 # Create project directory if it doesn't exist
389 project_path.mkdir(parents=True, exist_ok=True)
390
391 # Create configuration based on language
392 config = DoxygenConfig(
393 project_name=project_name,
394 output_directory=str(project_path / "docs"),
395 input_paths=[str(project_path)],
396 recursive=include_subdirs,
397 extract_private=extract_private
398 )
399
400 # Language-specific optimizations
401 if language == "c":
402 config.optimize_output_for_c = True
403 config.file_patterns = ["*.c", "*.h"]
404 elif language == "cpp":
405 config.file_patterns = ["*.cpp", "*.hpp", "*.cc", "*.hh", "*.cxx", "*.hxx"]
406 elif language == "python":
407 config.optimize_output_java = True # Python uses Java-style optimization
408 config.file_patterns = ["*.py"]
409 elif language == "php":
410 config.file_patterns = ["*.php", "*.php3", "*.inc"]
411 elif language == "java":
412 config.optimize_output_java = True
413 config.file_patterns = ["*.java"]
414 elif language == "csharp":
415 config.file_patterns = ["*.cs"]
416 elif language == "javascript":
417 config.file_patterns = ["*.js", "*.jsx", "*.ts", "*.tsx"]
418
419 # Save configuration
420 doxyfile_path = project_path / "Doxyfile"
421 with open(doxyfile_path, 'w', encoding='utf-8') as f:

430 Language: {language}
431 Configuration: {doxyfile_path}

4423. Run 'generate_documentation' to create docs

512 """Scan project structure and identify files"""

515 if not project_path.exists():

541

565 async def _suggest_file_patterns(self, args: Dict[str, Any]) -> Sequence[TextContent]:
566 """
567 @brief Suggest appropriate file patterns for a project
568 @param args Dictionary containing analysis parameters
569 @return Sequence of TextContent with pattern suggestions
570
571 @details Analyzes the project directory to identify file types and
572 suggests appropriate FILE_PATTERNS for Doxygen configuration based on
573 the actual files found in the project.
574 """
575 project_path = Path(args["project_path"])
576 primary_language = args.get("primary_language", "")
577 include_tests = args.get("include_tests", False)
578 include_examples = args.get("include_examples", True)
579
580 if not project_path.exists():
581 return [TextContent(type="text", text=f" Project path does not exist: {project_path}")]
582
583 try:
584 # Analyze actual files in the project
585 extensions = {}
586 for file_path in project_path.rglob("*"):
587 if file_path.is_file():
588 ext = file_path.suffix.lower()
589 if ext:
590 extensions[ext] = extensions.get(ext, 0) + 1
591
592 # Language-specific pattern suggestions
593 language_patterns = {
594 "c": ["*.c", "*.h"],
595 "cpp": ["*.cpp", "*.cxx", "*.cc", "*.C", "*.hpp", "*.hxx", "*.hh", "*.H"],
596 "python": ["*.py", "*.pyx", "*.pyi"],
597 "java": ["*.java"],
598 "php": ["*.php", "*.php3", "*.inc"],
599 "javascript": ["*.js", "*.jsx", "*.ts", "*.tsx"],
600 "csharp": ["*.cs"],
601 "go": ["*.go"],
602 "rust": ["*.rs"]
603 }
604
605 # Build suggestions based on found files and language
606 suggested_patterns = []
607
608 if primary_language.lower() in language_patterns:
609 suggested_patterns.extend(language_patterns[primary_language.lower()])
610
611 # Add patterns based on actual files found
612 common_source_extensions = {
613 ".c", ".cpp", ".cxx", ".cc", ".h", ".hpp", ".hxx", ".hh",
614 ".py", ".java", ".php", ".js", ".jsx", ".ts", ".tsx",
615 ".cs", ".go", ".rs", ".rb", ".pl", ".sh"
616 }
617
618 for ext in extensions:
619 if ext in common_source_extensions:
620 pattern = f"*{ext}"
621 if pattern not in suggested_patterns:
622 suggested_patterns.append(pattern)
623
624 # Optional patterns
625 optional_patterns = []
626 if include_examples and any(ext in [".md", ".txt", ".rst"] for ext in extensions):
627 optional_patterns.extend(["*.md", "*.txt", "*.rst"])
628
629 if include_tests:
630 optional_patterns.extend(["test_*.py", "*_test.cpp", "Test*.java"])
631
632 result_text = f""" File Pattern Suggestions for: {project_path}
633
634 Recommended Patterns:
635{chr(10).join(f" {pattern}" for pattern in suggested_patterns)}
636
637 Found Extensions:
638{chr(10).join(f" {ext}: {count} files" for ext, count in sorted(extensions.items(), key=lambda x: x[1], reverse=True)[:10])}
639"""
640
641 if optional_patterns:
642 result_text += f"\n Optional Patterns:\n{chr(10).join(f' {pattern}' for pattern in optional_patterns)}"
643
644 result_text += f"\n\n Add these patterns to your Doxyfile:\nFILE_PATTERNS = {' '.join(suggested_patterns)}"
645
646 return [TextContent(type="text", text=result_text)]
647
648 except Exception as e:
649 return [TextContent(type="text", text=f" Error analyzing patterns: {str(e)}")]
